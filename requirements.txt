# ==============================================================================
# ComfyUI ROCm 7.2.4 Virtual Environment Requirements
# ==============================================================================
#
# INSTALLATION STEPS:
#   1. Install ROCm system libraries first (see below):
#        sudo bash scripts/upgrade_rocm.sh
#   2. Activate venv & install pip packages:
#        source venv/bin/activate
#   3. Install pip packages:
#        pip install -r requirements.txt
#
# ==============================================================================
# SYSTEM PREREQUISITES — Install via: sudo dnf install -y <package>
# (Not a pip package — must be installed on the system before running pip install)
# ==============================================================================
#
# ROCm Runtime (required):
#   rocm-runtime
#   hip-runtime-amd
#   rocm-hip-runtime
#   rocm-smi-lib
#   rocminfo
#
# ROCm Math & Sparse Libraries (required for PyTorch ROCm):
#   hipsparse          # libhipsparse.so.4  ← torch fails to load without this
#   hipsparse7.2.4     # versioned alias
#   rocsparse          # backend hipsparse
#   rocsparse7.2.4
#   rocblas            # BLAS for GPU
#   rocblas7.2.4
#   hipblas            # HIP BLAS wrapper
#   hipblas7.2.4
#   hipblaslt          # hipBLASLt (flash attention, transformer ops)
#   hipblaslt7.2.4
#   rocfft             # FFT ops
#   rocfft7.2.4
#   hipsolver          # Linear solver
#   hipsolver7.2.4
#
# ROCm AI/Compute Libraries:
#   miopen-hip         # DNN primitives (conv, BN, etc.)
#   miopen-hip7.2.4
#   rocm-dev           # Headers & build tools (optional, for compiling)
#   comgr              # Code object manager
#   comgr7.2.4
#
# One-liner to install all:
#   sudo dnf install -y \
#     hipsparse hipsparse7.2.4 \
#     rocsparse rocsparse7.2.4 \
#     rocblas rocblas7.2.4 \
#     hipblas hipblas7.2.4 \
#     hipblaslt hipblaslt7.2.4 \
#     rocfft rocfft7.2.4 \
#     hipsolver hipsolver7.2.4 \
#     miopen-hip miopen-hip7.2.4 \
#     rocm-runtime hip-runtime-amd rocm-hip-runtime \
#     rocm-smi-lib rocminfo comgr comgr7.2.4
#
# Or run: sudo bash scripts/upgrade_rocm.sh
#
# ==============================================================================
# DEVELOPMENT, BUILD & UTILITY TOOLS — Install via: sudo dnf install -y <package>
# ==============================================================================
#
# Toolchain & Build Utilities:
#   nodejs npm             # Required to build the frontend & run npx tauri
#   rustup / rust cargo    # Required to compile the Rust back-end
#   rpm-build              # Required by Tauri to package the RPM installer
#
# Extraction & Local Installation:
#   rpm2cpio cpio          # Required by install_local.sh to extract the RPM package
#   desktop-file-utils     # Provides update-desktop-database
#   gtk3                   # Provides gtk-update-icon-cache
#
# Model Downloading:
#   wget                   # Required by download_models.sh to fetch AI models
#   curl                   # Required by download_models.sh to check URL validity
#   coreutils              # Provides numfmt used by download_models.sh
#
# CI/Linting/Quality Assurance:
#   jq                     # Required by ci-local.sh to validate tauri.conf.json
#   shellcheck             # Required by ci-local.sh to lint bash scripts
#   cargo-audit            # Install via: cargo install cargo-audit --locked
#
# ==============================================================================
