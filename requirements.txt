# ==============================================================================
# ComfyUI ROCm 7.2.4 Virtual Environment Requirements
# ==============================================================================
#
# LANGKAH INSTALL:
#   1. Install system libraries ROCm dulu (lihat bawah):
#        sudo bash scripts/upgrade_rocm.sh
#   2. Aktifkan venv & install pip packages:
#        source venv/bin/activate
#        pip install -r requirements.txt
#
# ==============================================================================
# SYSTEM PREREQUISITES — Install via: sudo dnf install -y <package>
# (Bukan pip package — harus diinstall di sistem sebelum pip install)
# ==============================================================================
#
# ROCm Runtime (wajib):
#   rocm-runtime
#   hip-runtime-amd
#   rocm-hip-runtime
#   rocm-smi-lib
#   rocminfo
#
# ROCm Math & Sparse Libraries (wajib untuk PyTorch ROCm):
#   hipsparse          # libhipsparse.so.4  ← tanpa ini torch gagal load
#   hipsparse7.2.4     # versioned alias
#   rocsparse          # backend hipsparse
#   rocsparse7.2.4
#   rocblas            # BLAS untuk GPU
#   rocblas7.2.4
#   hipblas            # HIP BLAS wrapper
#   hipblas7.2.4
#   hipblaslt          # hipBLASLt (flash attention, transformer ops)
#   hipblaslt7.2.4
#   rocfft             # FFT ops
#   rocfft7.2.4
#   hipsolver          # Linear solver
#   hipsolver7.2.4
#
# ROCm AI/Compute Libraries:
#   miopen-hip         # DNN primitives (conv, BN, etc.)
#   miopen-hip7.2.4
#   rocm-dev           # Headers & build tools (opsional, untuk compile)
#   comgr              # Code object manager
#   comgr7.2.4
#
# One-liner install semua:
#   sudo dnf install -y \
#     hipsparse hipsparse7.2.4 \
#     rocsparse rocsparse7.2.4 \
#     rocblas rocblas7.2.4 \
#     hipblas hipblas7.2.4 \
#     hipblaslt hipblaslt7.2.4 \
#     rocfft rocfft7.2.4 \
#     hipsolver hipsolver7.2.4 \
#     miopen-hip miopen-hip7.2.4 \
#     rocm-runtime hip-runtime-amd rocm-hip-runtime \
#     rocm-smi-lib rocminfo comgr comgr7.2.4
#
# Atau jalankan: sudo bash scripts/upgrade_rocm.sh
#
# ==============================================================================
